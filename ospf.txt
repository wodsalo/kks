Алгоритм маршрутизации с учетом состояния линии ospf

Настройка маршутизаторов состоит из несколькиъ частей

1 пункт - присвоение ip адресов интерфейсам loopback
R0(config)#int loopback 0
ip address 192.168.100.1 255.255.255.255
...

2 пунск - включение процесса OSPF
R0(config)#router ospf 1

3 пункт - обьявление сетей
R0(config-router)#network 192.168.10.0 0.0.0.127 area 0
R0(config-router)#network 192.168.1.0 0.0.0.255 area 0
R0(config-router)#network 192.168.4.0 0.0.0.3 area 0
R0(config-router)#network 10.0.0.0 0.255.255.255 area 0
R0(config-router)#network 11.0.0.0 0.255.255.255 area 0
